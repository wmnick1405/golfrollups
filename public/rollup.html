function shuffle(array) { for (let i = array.length - 1; i > 0; i--) { let j = Math.floor(Math.random() * (i + 1)); let temp = array[i]; array[i] = array[j]; array[j] = temp; } return array; } // Build groups of 3 and 4 based on the total number of players function buildGroups(players) { let total = players.length; if (total === 0) { alert("No golfers available for this date"); return []; } if (total < 3) { alert("Not enough golfers to form a group (minimum 3 required)"); return []; } let fours = 0; let threes = 0; // Logic to ensure only groups of 3 and 4 are created if (total % 4 === 0) { fours = total / 4; } else if (total % 4 === 3) { fours = Math.floor(total / 4); threes = 1; } else if (total % 4 === 2) { // e.g., 6 or 10 players -> split into 3s fours = Math.floor(total / 4) - 1; threes = 2; } else if (total % 4 === 1) { // e.g., 5, 9, or 13 players -> split into 3s fours = Math.floor(total / 4) - 2; threes = 3; } let groups = []; // Extract the players into the calculated group sizes for (let i = 0; i < fours; i++) { groups.push(players.splice(0, 4)); } for (let i = 0; i < threes; i++) { groups.push(players.splice(0, 3)); } return groups; } function chooseBooker(group) { group.sort((a, b) => { if (a.booking_count !== b.booking_count) { return a.booking_count - b.booking_count; } return new Date(a.last_booked) - new Date(b.last_booked); }); return group; } async function generate() { const date = document.getElementById('date').value; if (!date) { alert("Please select a date first"); return; } // --- 1. DUPLICATE CHECK (Check before generating) --- const checkResponse = await fetch(`/api/rollups/find?date=${date}`); if (checkResponse.ok) { const existing = await checkResponse.json(); alert("A rollup already exists for this date."); if (confirm("Would you like to view the existing booking sheet?")) { window.location.href = "/bookingsheet.html?id=" + existing._id; } return; // Stop the function here } // --- 2. GET AVAILABLE PLAYERS --- const response = await fetch(`/api/available?date=${date}`); if (response.status === 401) { alert("Session expired. Please log in."); window.location.href = "/"; return; } const players = await response.json(); shuffle(players); // --- 3. BUILD GROUPS --- const groups = buildGroups(players); if (groups.length === 0) return; const div = document.getElementById('groups'); div.innerHTML = ""; let saveGroups = []; groups.forEach((group, i) => { group = chooseBooker(group); const g = document.createElement('div'); g.className = "group"; let html = `Group ${i + 1}
`; let saveGroup = []; group.forEach((p, index) => { if (index === 0) { fetch("/api/booker/" + p._id, { method: "POST" }); html += "" + p.name + " (Booker)
"; saveGroup.push({ golfer_id: p._id, name: p.name, booker: true }); } else { html += p.name + "
"; saveGroup.push({ golfer_id: p._id, name: p.name, booker: false }); } }); saveGroups.push(saveGroup); g.innerHTML = html; div.appendChild(g); }); // --- 4. SAVE THE NEW ROLLUP --- const saveResponse = await fetch('/api/rollups', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ date: date, groups: saveGroups }) }); const result = await saveResponse.json(); if (result.success) { console.log("Rollup saved successfully"); } } // Nav Placeholder loader fetch('nav.html') .then(response => response.text()) .then(data => { document.getElementById('nav-placeholder').innerHTML = data; });
